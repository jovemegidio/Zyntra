Como Usar

Copie e adapte este código para suas páginas

<!-- 1. Incluir CSS e JS -->
<link rel="stylesheet" href="/css/modal-unified.css">
<script src="/js/modal-unified.js"></script>

<!-- 2. Estrutura do Modal -->
<div class="modal-unified" id="meu-modal">
  <div class="modal-unified-content modal-md">
    <div class="modal-unified-header">
      <h2><i class="fas fa-cog"></i> Título</h2>
      <button class="modal-unified-close" data-close-modal>×</button>
    </div>
    <div class="modal-unified-body">
      <!-- Conteúdo aqui -->
    </div>
    <div class="modal-unified-footer">
      <button class="btn btn-secondary" data-close-modal>Cancelar</button>
      <button class="btn btn-primary">Salvar</button>
    </div>
  </div>
</div>

<!-- 3. JavaScript -->
ModalUnified.open('meu-modal'); // Abrir
ModalUnified.close('meu-modal'); // Fechar

// Confirmação
const confirmed = await ModalUnified.confirm({
  title: 'Excluir?',
  message: 'Esta ação é irreversível.',
  type: 'danger'
});
